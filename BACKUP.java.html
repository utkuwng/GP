        // Button to submit
        Button submitButton = new Button("Gửi");
        submitButton.setFont(Font.font("Arial", 16));
        submitButton.setOnAction(e -> {
            // Lấy dữ liệu từ các trường
            String name = nameField.getText();
            String message = messageArea.getText();
            String area = areaArea.getText();
            // Kiểm tra nếu có trường nào còn trống
            if (name.isEmpty() || message.isEmpty() || area.isEmpty()) {
                // Hiển thị thông báo lỗi nếu có trường trống
                showAlert(Alert.AlertType.ERROR, "Lỗi", "Vui lòng điền đầy đủ thông tin.");
            } else {
                insertMessage(name, area, message);
                showAlert(Alert.AlertType.INFORMATION, "Thông báo", "Bạn đã gửi thông điệp thành công.");
            }
            nameField.clear();
            messageArea.clear();
            areaArea.clear();
        });
        formLayout.getChildren().addAll(nameLabel, nameField, messageLabel, messageArea, areaLabel, areaArea, submitButton);
        return formLayout;
    }
    private void showAlert(Alert.AlertType alertType, String title, String message) {
        Alert alert = new Alert(alertType);
        alert.setTitle(title);
        alert.setHeaderText(null);
        alert.setContentText(message);
        alert.showAndWait();
    }
    private static ScrollPane getBorderPane(HBox header, VBox createFooter) {
        // Tạo ImageView cho hình ảnh nền
        Image backgroundImage = new Image("canh.gif");
        ImageView bgImageView = new ImageView(backgroundImage);
        bgImageView.setFitWidth(1400);  // Kích thước chiều rộng của hình ảnh
        bgImageView.setFitHeight(700); // Kích thước chiều cao của hình ảnh
        bgImageView.setPreserveRatio(false);

        // Tạo tiêu đề (đè lên ảnh nền)
        Label titleLabel = new Label("GREENPATH");
        titleLabel.setStyle("-fx-font-family: Roboto;" +
                "-fx-font-size:100 px;" +
                "-fx-font-weight: bold;" +
                "-fx-text-fill: white;"); // Văn bản màu trắng để nổi bật trên nền
        StackPane.setAlignment(titleLabel, Pos.CENTER); // Đặt vị trí titleLabel ở chính giữa

        // Tạo StackPane để chồng ảnh nền và tiêu đề
        StackPane stackPane = new StackPane();
        stackPane.getChildren().addAll(bgImageView, titleLabel); // Ảnh nền phía dưới, tiêu đề phía trên
        stackPane.setPrefSize(1400, 700); // Đảm bảo kích thước StackPane khớp với ảnh

        // Tạo VBox làm root layout
        VBox root = new VBox();
        root.getChildren().addAll(header, stackPane, createFooter);

        // Thêm ScrollPane để cuộn nội dung
        ScrollPane scrollPane = new ScrollPane(root);
        scrollPane.setFitToWidth(true); // Làm cho nội dung tự động co giãn theo chiều rộng
        scrollPane.setVbarPolicy(ScrollPane.ScrollBarPolicy.ALWAYS); // Luôn hiển thị thanh cuộn dọc
        scrollPane.setHbarPolicy(ScrollPane.ScrollBarPolicy.NEVER); // Không hiển thị thanh cuộn ngang
        // Đảm bảo ScrollPane hiển thị từ đầu (vị trí đầu trang)
        scrollPane.setVvalue(0.0); // Cuộn về đầu theo chiều dọc

        // Trả về ScrollPane chứa VBox
        return scrollPane;
    }

    private void cycleImages(HBox studentGallery) {

        String[][] imageSets = {
                {"2.jpg", "3.jpg", "6.jpg", "Nguyễn Song Nhi", "Châu Quỳnh Như", "Huỳnh Ngọc Nga"},
                {"7.jpg", "9.jpg", "12.jpg", "Nguyễn Lê Quỳnh", "Nguyễn Hà My", "Lê Huỳnh Như"}
        };


        String[] currentSet = imageSets[imageSetIndex];

        for (int i = 0; i < 3; i++) {
            VBox studentCard = (VBox) studentGallery.getChildren().get(i);
            ImageView imageView = (ImageView) studentCard.getChildren().get(0);
            imageView.setImage(new Image(currentSet[i]));
            Label nameLabel = (Label) studentCard.getChildren().get(1);
            nameLabel.setText(currentSet[i + 3]);
        }

        // Update the image set index
        imageSetIndex = (imageSetIndex + 1) % imageSets.length;
    }

    private Button createTransparentButton(String text) {
        Button button = new Button(text);
        button.setStyle("-fx-background-color: transparent; " +
                "-fx-text-fill: white; " +
                "-fx-underline: true; " + // Gạch chân
                "-fx-font-size: 18px; " +
                "-fx-border-color: transparent; " + // Khung trong suốt
                "-fx-cursor: hand;"); // Thêm con trỏ chuột khi hover
        button.setOnMouseEntered(e -> button.setStyle("-fx-background-color: transparent; " +
                "-fx-text-fill: #d1dead; " + // Đổi màu chữ khi hover
                "-fx-underline: true; " +
                "-fx-font-size: 18px; " +
                "-fx-border-color: transparent;"));
        button.setOnMouseExited(e -> button.setStyle("-fx-background-color: transparent; " +
                "-fx-text-fill: white; " +
                "-fx-underline: true; " +
                "-fx-font-size: 18px; " +
                "-fx-border-color: transparent;"));
        return button;
    }
    private ImageView createSocialIcon(String imagePath, String link) {
        ImageView icon = new ImageView(new Image(imagePath));
        icon.setFitHeight(50);
        icon.setFitWidth(50);

        // Thêm sự kiện vào icon để mở link khi nhấn
        icon.setOnMouseClicked(event -> {
            try {
                URI uri = new URI(link); // Tạo đối tượng URI từ link
                Desktop.getDesktop().browse(uri); // Mở link trong trình duyệt mặc định
            } catch (Exception e) {
                e.printStackTrace(); // Xử lý ngoại lệ nếu có lỗi
            }
        });
        return icon;
    }
